    #include <iostream>
    #include <cstring>

    using namespace std;

    char c[100], b[] = { 'a', 'e', 'o', 'u' };
    int nr, ok, i;

    int main()
    {
        cin.getline(c, 100);	
        for (i = 0; i < strlen(c); i++)
        {
            if (strchr(b, c[i]))	
              ok++;
          else			
            {
                if (c[i] == 'i')	
                  nr++;
            }
        }
        if (ok != 0 || nr == strlen(c))	
          cout << "NU";
        else
          cout << "DA";
        return 0;
    }
